  ```python
  from odoo import http
  from odoo.http import request

  class WujiaLegacyRedirects(http.Controller):
      @http.route('/portal/purchase_history', type='http', auth='user', website=True)
      def legacy_purchase_history(self, **kw):
          return request.redirect('/portal/purchase-history', code=301)

      @http.route('/portal/return-request-list', type='http', auth='user', website=True)
      def legacy_return_list(self, **kw):
          return request.redirect('/portal/return', code=301)

      @http.route('/portal/exam-registration', type='http', auth='user', website=True)
      def legacy_exam_register(self, **kw):
          return request.redirect('/portal/exam/my', code=301)
  ```
