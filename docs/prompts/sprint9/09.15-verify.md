Pattern theo `test_sprint5.py` (đã tồn tại). Cấu trúc:

```python
"""Sprint 9 smoke test — 13 portal route + 13 CSS assertion."""
import requests, re, sys

BASE = "http://localhost:8069"
LOGIN = "admin"
PASSWORD = "1"

passed, failed = 0, 0

def check(name, cond, detail=""):
    global passed, failed
    if cond:
        print(f"PASS: {name}")
        passed += 1
    else:
        print(f"FAIL: {name} — {detail}")
        failed += 1

# 1) Login + get session cookie
session = requests.Session()
# ... auth flow

# 2) Route checks (13 portal route)
ROUTES = [
    "/portal", "/portal/order", "/portal/purchase-history",
    "/portal/delivery", "/portal/return", "/portal/notification",
    "/portal/exam", "/portal/knowledge", "/portal/support",
    "/portal/info-request", "/portal/reports/orders",
    "/portal/franchise-information", "/portal/profile",
]
for r in ROUTES:
    resp = session.get(BASE + r)
    check(f"route {r}", resp.status_code == 200, f"got {resp.status_code}")

# 3) CSS assertions (load /portal HTML, extract CSS file URLs, fetch + assert)
home = session.get(BASE + "/portal").text
css_urls = re.findall(r'href="(/web/assets/[^"]+\.css)"', home)
all_css = "".join(session.get(BASE + u).text for u in css_urls)

# UI-01: active menu BG primary
check("UI-01 active BG primary",
      "background-color: var(--wujia-primary)" in all_css and ".active" in all_css)

# UI-02: sidebar-header user-pic removed from template (check HTML, not CSS)
check("UI-02 no user-pic", 'class="user-pic"' not in home)

# UI-03: Current Store block present
check("UI-03 current store block", "wujia-current-store" in home or "[H" in home)

# ... 10 more assertions, 1 per issue ...

# 4) 301 redirects (Sprint 9.14)
r1 = session.get(BASE + "/portal/purchase_history", allow_redirects=False)
check("9.14 redirect purchase_history → purchase-history",
      r1.status_code == 301 and "/portal/purchase-history" in r1.headers.get("Location", ""))

print(f"\n{passed} PASS / {failed} FAIL")
sys.exit(0 if failed == 0 else 1)
```
